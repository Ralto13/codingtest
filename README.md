```javscript
const str = 'Hello';
str.length

// 2
str.replace(/ll/,'tt')
// Hetto
str.split('')
// [ 'H', 'e', 'l', 'l', 'o' ]
str.repeat(2)
// HelloHello
str.padStart(10); //
'     Hello'
str.padEnd(10); //
'Hello     '
```
